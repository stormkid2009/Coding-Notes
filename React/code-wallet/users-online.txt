Title: " show online users "
with the help of filter and map methods


<Code>

export default class App extends React.Component{
  constructor(props){
    super();
    this.state={
      users:[
          {userName:"Ahmed",
            online:true},
          {userName:"Ali",
            online:false},
          { userName:"Farah",
             online:true},
          { userName:"Maggie",
              online:true},
          { userName:"Joe",
              online:false}
      ]
    }

  }
  render(){
  const  getUsers=this.state.users.filter(person=>person.online);
  const userArr=getUsers.map(person=><li key={person.userName+1}>{person.userName}</li>);
  return(<div>
  <h1>Online Users are {userArr.length}</h1>
  <ul>{userArr}</ul>
  </div>);
  }
  
}
</Code>
